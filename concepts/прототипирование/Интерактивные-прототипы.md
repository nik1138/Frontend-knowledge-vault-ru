# Интерактивные прототипы в фронтенд-разработке

Интерактивные прототипы - это мощный инструмент для демонстрации поведения интерфейса, проверки пользовательских сценариев и получения обратной связи до начала полноценной разработки.

## Что такое интерактивные прототипы

Интерактивные прототипы - это модели интерфейса, которые имитируют реальное взаимодействие пользователя с приложением. В отличие от статичных макетов, они включают в себя:
- Анимации и переходы
- Обработку событий
- Логику взаимодействия
- Пользовательские сценарии

## Преимущества интерактивных прототипов

### 1. Реалистичная демонстрация
Интерактивные прототипы позволяют увидеть, как будет вести себя интерфейс в реальных условиях.

### 2. Тестирование пользовательских сценариев
Можно протестировать различные пути взаимодействия пользователя с интерфейсом.

### 3. Раннее выявление проблем
Проблемы с юзабилити можно выявить на ранних этапах, до начала разработки.

### 4. Улучшенная коммуникация
Интерактивные прототипы помогают лучше объяснить идеи команде и заинтересованным сторонам.

### 3. Реализация сложной логики взаимодействия

```javascript
// Прототип формы с валидацией и динамическими элементами
class FormPrototype {
  constructor(formId) {
    this.form = document.getElementById(formId);
    this.state = {};
    this.init();
  }

  init() {
    this.form.addEventListener('submit', (e) => this.handleSubmit(e));
    this.setupValidation();
    this.setupDynamicFields();
  }

  setupValidation() {
    const inputs = this.form.querySelectorAll('input, textarea, select');
    inputs.forEach(input => {
      input.addEventListener('blur', () => this.validateField(input));
      input.addEventListener('input', () => this.clearError(input));
    });
  }

  validateField(field) {
    const value = field.value.trim();
    const fieldName = field.name;
    let isValid = true;
    let errorMessage = '';

    // Простая валидация
    if (field.hasAttribute('required') && !value) {
      isValid = false;
      errorMessage = 'Это поле обязательно';
    } else if (field.type === 'email' && !this.isValidEmail(value)) {
      isValid = false;
      errorMessage = 'Введите корректный email';
    }

    if (!isValid) {
      this.showError(field, errorMessage);
    }

    return isValid;
  }

  isValidEmail(email) {
    const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return re.test(email);
  }

  showError(field, message) {
    field.classList.add('error');
    let errorElement = field.parentNode.querySelector('.error-message');
    if (!errorElement) {
      errorElement = document.createElement('div');
      errorElement.className = 'error-message';
      errorElement.style.color = 'red';
      errorElement.style.fontSize = '14px';
      field.parentNode.appendChild(errorElement);
    }
    errorElement.textContent = message;
  }

  clearError(field) {
    field.classList.remove('error');
    const errorElement = field.parentNode.querySelector('.error-message');
    if (errorElement) {
      errorElement.remove();
    }
  }

  setupDynamicFields() {
    // Добавление динамических полей
    const addButton = this.form.querySelector('.add-field');
    if (addButton) {
      addButton.addEventListener('click', () => this.addField());
    }
  }

  addField() {
    const container = this.form.querySelector('.dynamic-fields');
    const fieldCount = container.querySelectorAll('.dynamic-field').length + 1;
    
    const fieldDiv = document.createElement('div');
    fieldDiv.className = 'dynamic-field';
    fieldDiv.innerHTML = `
      <label>Динамическое поле ${fieldCount}:</label>
      <input type="text" name="dynamic_field_${fieldCount}" required>
      <button type="button" class="remove-field">Удалить</button>
    `;
    
    container.appendChild(fieldDiv);
    
    // Добавить обработчик для удаления
    fieldDiv.querySelector('.remove-field').addEventListener('click', () => {
      fieldDiv.remove();
    });
  }

  handleSubmit(e) {
    e.preventDefault();
    
    // Проверить валидность всех полей
    const inputs = this.form.querySelectorAll('input, textarea, select');
    let isFormValid = true;
    
    inputs.forEach(input => {
      if (!this.validateField(input)) {
        isFormValid = false;
      }
    });
    
    if (isFormValid) {
      // Симуляция отправки формы
      this.showLoading();
      setTimeout(() => {
        this.hideLoading();
        alert('Форма успешно отправлена!');
      }, 2000);
    } else {
      alert('Пожалуйста, исправьте ошибки в форме');
    }
  }

  showLoading() {
    const submitButton = this.form.querySelector('button[type="submit"]');
    submitButton.disabled = true;
    submitButton.innerHTML = '<div class="loading"></div> Отправка...';
  }

  hideLoading() {
    const submitButton = this.form.querySelector('button[type="submit"]');
    submitButton.disabled = false;
    submitButton.textContent = 'Отправить';
  }
}

// Инициализация прототипа формы
document.addEventListener('DOMContentLoaded', () => {
  new FormPrototype('prototypeForm');
});
```

## Продвинутые интерактивные элементы

### 1. Динамические списки с фильтрацией

```html
<div class="searchable-list">
  <input type="text" id="searchInput" placeholder="Поиск...">
  <ul id="list">
    <li data-category="frontend">HTML</li>
    <li data-category="frontend">CSS</li>
    <li data-category="frontend">JavaScript</li>
    <li data-category="backend">Node.js</li>
    <li data-category="backend">Python</li>
    <li data-category="database">MongoDB</li>
    <li data-category="database">PostgreSQL</li>
  </ul>
</div>

<script>
  document.getElementById('searchInput').addEventListener('input', function() {
    const searchTerm = this.value.toLowerCase();
    const listItems = document.querySelectorAll('#list li');
    
    listItems.forEach(item => {
      const text = item.textContent.toLowerCase();
      if (text.includes(searchTerm)) {
        item.style.display = 'block';
      } else {
        item.style.display = 'none';
      }
    });
  });
</script>
```

### 2. Слайдер с интерактивными элементами

```html
<div class="slider-container">
  <div class="slider" id="slider">
    <div class="slide active">
      <h3>Слайд 1</h3>
      <p>Содержимое первого слайда</p>
    </div>
    <div class="slide">
      <h3>Слайд 2</h3>
      <p>Содержимое второго слайда</p>
    </div>
    <div class="slide">
      <h3>Слайд 3</h3>
      <p>Содержимое третьего слайда</p>
    </div>
  </div>
  <div class="slider-controls">
    <button id="prev">Предыдущий</button>
    <div class="dots">
      <span class="dot active" data-slide="0"></span>
      <span class="dot" data-slide="1"></span>
      <span class="dot" data-slide="2"></span>
    </div>
    <button id="next">Следующий</button>
  </div>
</div>

<script>
  let currentSlide = 0;
  const slides = document.querySelectorAll('.slide');
  const dots = document.querySelectorAll('.dot');
  const prevBtn = document.getElementById('prev');
  const nextBtn = document.getElementById('next');

  function showSlide(n) {
    slides.forEach(slide => slide.classList.remove('active'));
    dots.forEach(dot => dot.classList.remove('active'));
    
    currentSlide = (n + slides.length) % slides.length;
    slides[currentSlide].classList.add('active');
    dots[currentSlide].classList.add('active');
  }

  prevBtn.addEventListener('click', () => showSlide(currentSlide - 1));
  nextBtn.addEventListener('click', () => showSlide(currentSlide + 1));

  dots.forEach((dot, index) => {
    dot.addEventListener('click', () => showSlide(index));
  });
</script>
```

## Тестирование интерактивных прототипов

### 1. Пользовательские сценарии

Создавайте сценарии использования, чтобы протестировать прототип:
- Регистрация нового пользователя
- Поиск товара
- Оформление заказа
- Просмотр профиля

### 2. Тестирование на разных устройствах

Убедитесь, что интерактивные элементы работают корректно на разных размерах экранов:
- Десктоп
- Планшет
- Мобильные устройства

### 3. Тестирование доступности

- Проверьте работу с клавиатуры
- Убедитесь, что элементы доступны для скринридеров
- Проверьте контрастность цветов

## Лучшие практики

### 1. Сосредоточьтесь на ключевых сценариях
Создавайте интерактивность только для основных пользовательских путей.

### 2. Используйте реальные данные
Для более реалистичного прототипа используйте данные, близкие к реальным.

### 3. Обеспечьте обратную связь
Пользователь должен понимать, что происходит с интерфейсом (загрузка, успех, ошибка).

### 4. Поддерживайте простоту
Не перегружайте прототип сложной логикой - он должен быть понятным и быстрым в создании.
