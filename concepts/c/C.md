# C

C — это процедурный язык программирования общего назначения, разработанный в начале 1970-х годов Деннисом Ритчи. C известен своей эффективностью, гибкостью и близостью к машинному коду, что делает его идеальным для системного программирования и разработки операционных систем.

## Основные характеристики

### Низкоуровневый доступ
C предоставляет прямой доступ к памяти и аппаратным ресурсам, что делает его очень эффективным.

```c
// Прямой доступ к памяти через указатели
int x = 42;
int *ptr = &x;  // Указатель на переменную x
printf("Value: %d\n", *ptr);  // Разыменование указателя
```

### Простота и эффективность
C имеет минимальный набор ключевых слов и конструкций, что делает его простым и эффективным.

```c
// Простая структура программы на C
#include <stdio.h>

int main() {
    printf("Hello, World!\n");
    return 0;
}
```

### Переносимость
Программы на C могут быть скомпилированы и запущены на различных платформах с минимальными изменениями.

```c
// Переносимый код
#ifdef _WIN32
    #include <windows.h>
#else
    #include <unistd.h>
#endif

void sleep_seconds(int seconds) {
#ifdef _WIN32
    Sleep(seconds * 1000);  // Windows использует миллисекунды
#else
    sleep(seconds);         // Unix использует секунды
#endif
}
```

## Основные концепции

### Структурное программирование
C построен на принципах структурного программирования с использованием функций.

```c
// Функции в C
#include <stdio.h>

// Объявление функции
int factorial(int n);

int main() {
    int num = 5;
    int result = factorial(num);
    printf("Factorial of %d is %d\n", num, result);
    return 0;
}

// Определение функции
int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);  // Рекурсия
}
```

### Указатели и работа с памятью
C предоставляет полный контроль над памятью через указатели.

```c
// Работа с указателями
#include <stdio.h>
#include <stdlib.h>

int main() {
    // Динамическое выделение памяти
    int *arr = (int*)malloc(5 * sizeof(int));
    
    // Инициализация массива
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 10;
    }
    
    // Использование массива
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d\n", i, arr[i]);
    }
    
    // Освобождение памяти
    free(arr);
    return 0;
}
```

### Структуры и объединения
C поддерживает пользовательские типы данных через структуры.

```c
// Структуры в C
#include <stdio.h>
#include <string.h>

// Определение структуры
struct Person {
    char name[50];
    int age;
    float height;
};

// Объединения
union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    // Создание и инициализация структуры
    struct Person person1;
    strcpy(person1.name, "John");
    person1.age = 30;
    person1.height = 5.9;
    
    printf("Name: %s\n", person1.name);
    printf("Age: %d\n", person1.age);
    printf("Height: %.1f\n", person1.height);
    
    // Использование объединения
    union Data data;
    data.i = 10;
    printf("data.i: %d\n", data.i);
    
    data.f = 220.5;
    printf("data.f: %f\n", data.f);
    
    return 0;
}
```

## Преимущества C

1. **Эффективность** — Близость к машинному коду обеспечивает высокую производительность
2. **Контроль** — Полный контроль над памятью и аппаратными ресурсами
3. **Переносимость** — Программы могут быть скомпилированы на различных платформах
4. **Простота** — Минимальный набор ключевых слов и конструкций
5. **Основа для других языков** — Многие современные языки основаны на синтаксисе C

## В других технологиях

### В [[systems-programming/Системное программирование]]
C широко используется для системного программирования:

```c
// Пример системного программирования
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>

int main() {
    pid_t pid = fork();  // Создание нового процесса
    
    if (pid == 0) {
        // Дочерний процесс
        printf("Child process\n");
        exit(0);
    } else if (pid > 0) {
        // Родительский процесс
        printf("Parent process\n");
        wait(NULL);  // Ожидание завершения дочернего процесса
    } else {
        // Ошибка
        perror("fork failed");
        exit(1);
    }
    
    return 0;
}
```

### В [[embedded-systems/Встраиваемые системы]]
C идеально подходит для программирования встраиваемых систем:

```c
// Пример для встраиваемой системы
#define LED_PIN 13

void setup() {
    // Инициализация пина как выхода
    pinMode(LED_PIN, OUTPUT);
}

void loop() {
    // Включение светодиода
    digitalWrite(LED_PIN, HIGH);
    delay(1000);  // Задержка 1 секунда
    
    // Выключение светодиода
    digitalWrite(LED_PIN, LOW);
    delay(1000);  // Задержка 1 секунда
}
```
