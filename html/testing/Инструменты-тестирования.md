# Инструменты тестирования HTML

Инструменты тестирования HTML - это программные решения, которые помогают разработчикам и тестировщикам проверять корректность, производительность, доступность и совместимость HTML-документов. В условиях российского рынка 2025 года, с учетом ограничений на западные технологии и развития отечественных решений, выбор подходящих инструментов тестирования становится особенно важным.

## Классификация инструментов тестирования HTML

### По типу тестирования

- **Валидация HTML**: Проверка соответствия стандартам HTML5
- **Доступность**: Проверка соответствия требованиям доступности
- **Кроссбраузерность**: Тестирование в разных браузерах
- **Производительность**: Анализ скорости и эффективности
- **Адаптивность**: Тестирование на разных устройствах
- **Безопасность**: Проверка уязвимостей

### По методу выполнения

- **Ручные инструменты**: Требующие непосредственного участия человека
- **Автоматизированные инструменты**: Работающие без участия человека
- **Гибридные инструменты**: Сочетающие автоматизацию и ручной контроль

### Пример использования валидатора в CI/CD

```javascript
// html-validator.js
const validator = require('html-validator');
const fs = require('fs');

async function validateHTMLFile(filePath) {
  const htmlContent = fs.readFileSync(filePath, 'utf8');
  
  const options = {
    data: htmlContent,
    format: 'text'
  };
  
  try {
    const report = await validator(options);
    console.log('Валидация прошла успешно для файла:', filePath);
    return true;
  } catch (error) {
    console.error('Ошибки валидации в файле:', filePath);
    console.error(error.message);
    return false;
  }
}

// Проверка всех HTML файлов в проекте
const htmlFiles = [
  './index.html',
  './about.html',
  './contact.html'
];

async function validateAllFiles() {
  let allValid = true;
  
  for (const file of htmlFiles) {
    const isValid = await validateHTMLFile(file);
    if (!isValid) {
      allValid = false;
    }
  }
  
  process.exit(allValid ? 0 : 1);
}

validateAllFiles();
```

## Инструменты тестирования доступности

### axe-core

Библиотека axe-core - один из самых популярных инструментов для проверки доступности:

```javascript
// Пример использования axe-core в тестах
const axe = require('axe-core');

// Проверка доступности на странице
axe.run(document, {
  runOnly: {
    type: 'tag',
    values: ['wcag2a', 'wcag2aa']
  }
}, (err, results) => {
  if (err) throw err;
  
  if (results.violations.length) {
    console.log('Найдены нарушения доступности:');
    results.violations.forEach(violation => {
      console.log(violation.help);
      console.log(violation.nodes.map(node => node.html));
    });
  } else {
    console.log('Нарушений доступности не найдено');
  }
});
```

### Автоматизированные инструменты

#### Selenium WebDriver

```javascript
// Пример теста с Selenium для нескольких браузеров
const { Builder, By, until } = require('selenium-webdriver');
const firefox = require('selenium-webdriver/firefox');
const chrome = require('selenium-webdriver/chrome');

async function testInBrowser(browserName) {
  let driver;
  
  switch(browserName) {
    case 'chrome':
      driver = await new Builder().forBrowser('chrome').build();
      break;
    case 'firefox':
      driver = await new Builder().forBrowser('firefox').build();
      break;
    case 'yandex':
      // Настройка для Яндекс.Браузера
      const options = new chrome.Options().setBinary('/path/to/yandex');
      driver = await new Builder().forBrowser('chrome').setChromeOptions(options).build();
      break;
    default:
      throw new Error('Неизвестный браузер');
  }
  
  try {
    await driver.get('http://localhost:3000');
    
    // Проверка наличия основных элементов
    const header = await driver.wait(until.elementLocated(By.tagName('header')), 5000);
    const nav = await driver.findElement(By.tagName('nav'));
    const main = await driver.findElement(By.tagName('main'));
    const footer = await driver.findElement(By.tagName('footer'));
    
    console.log(`${browserName}: Все основные элементы найдены`);
    
    // Проверка интерактивности
    const navLinks = await driver.findElements(By.css('nav a'));
    for (const link of navLinks) {
      const href = await link.getAttribute('href');
      console.log(`${browserName}: Найдена ссылка ${href}`);
    }
  } finally {
    await driver.quit();
  }
}

// Тестирование в разных браузерах
async function crossBrowserTest() {
  const browsers = ['chrome', 'firefox', 'yandex'];
  
  for (const browser of browsers) {
    try {
      await testInBrowser(browser);
    } catch (error) {
      console.error(`Ошибка при тестировании в ${browser}:`, error.message);
    }
  }
}

crossBrowserTest();
```

#### Puppeteer

```javascript
// Пример теста с Puppeteer
const puppeteer = require('puppeteer');

async function testHTMLStructure() {
  const browser = await puppeteer.launch({
    headless: true
  });
  
  const page = await browser.newPage();
  await page.goto('http://localhost:3000', { waitUntil: 'networkidle2' });
  
  // Проверка основных семантических элементов
  const hasHeader = await page.$eval('header', el => !!el);
  const hasNav = await page.$eval('nav', el => !!el);
  const hasMain = await page.$eval('main', el => !!el);
  const hasFooter = await page.$eval('footer', el => !!el);
  
  console.log('Структура HTML:');
  console.log('- Header:', hasHeader);
  console.log('- Nav:', hasNav);
  console.log('- Main:', hasMain);
  console.log('- Footer:', hasFooter);
  
  // Проверка форм
  const forms = await page.$$('.form, form');
  console.log(`Найдено форм: ${forms.length}`);
  
  // Проверка изображений с alt
  const imagesWithoutAlt = await page.$$eval('img:not([alt]), img[alt=""]', imgs => imgs.length);
  console.log(`Изображений без alt: ${imagesWithoutAlt}`);
  
  await browser.close();
}

testHTMLStructure();
```

## Инструменты тестирования производительности

### Lighthouse

Lighthouse - мощный инструмент для аудита производительности:

```javascript
// Использование Lighthouse CLI
const lighthouse = require('lighthouse');
const chromeLauncher = require('chrome-launcher');

async function runLighthouseAudit(url) {
  const chrome = await chromeLauncher.launch({chromeFlags: ['--headless']});
  const options = {
    logLevel: 'info',
    output: 'html',
    onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
    port: chrome.port,
  };
  
  const runnerResult = await lighthouse(url, options);
  
  // Сохранение отчета
  const reportHtml = runnerResult.report;
  require('fs').writeFileSync('lhreport.html', reportHtml);
  
  console.log('Отчет Lighthouse сохранен как lhreport.html');
  console.log('Производительность:', runnerResult.lhr.categories.performance.score * 100);
  console.log('Доступность:', runnerResult.lhr.categories.accessibility.score * 100);
  
  await chrome.kill();
}

runLighthouseAudit('http://localhost:3000');
```

### PageSpeed Insights API

```javascript
// Пример использования PageSpeed Insights API
const https = require('https');

function getPageSpeedData(url) {
  const apiUrl = `https://www.googleapis.com/pagespeedonline/v5/runPagespeed?url=${encodeURIComponent(url)}&strategy=mobile`;
  
  return new Promise((resolve, reject) => {
    https.get(apiUrl, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        try {
          const result = JSON.parse(data);
          resolve(result);
        } catch (error) {
          reject(error);
        }
      });
    }).on('error', reject);
  });
}

getPageSpeedData('http://localhost:3000')
  .then(result => {
    console.log('Оценка производительности:', result.lighthouseResult.categories.performance.score * 100);
    console.log('Оценка доступности:', result.lighthouseResult.categories.accessibility.score * 100);
  })
  .catch(console.error);
```

## Инструменты автоматизации тестирования

### Playwright

```javascript
// Пример теста с Playwright
const { chromium, firefox, webkit } = require('playwright');

async function runTests() {
  const browsers = [chromium, firefox, webkit];
  
  for (const browserType of browsers) {
    const browser = await browserType.launch();
    const page = await browser.newPage();
    
    await page.goto('http://localhost:3000');
    
    // Проверка HTML структуры
    const htmlStructure = {
      header: await page.isVisible('header'),
      nav: await page.isVisible('nav'),
      main: await page.isVisible('main'),
      footer: await page.isVisible('footer')
    };
    
    console.log(`${browserType.name()}:`, htmlStructure);
    
    await browser.close();
  }
}

runTests();
```

## Интеграция инструментов в CI/CD

### Пример конфигурации GitHub Actions

```yaml
name: HTML Testing
on: [push, pull_request]

jobs:
  html-validation:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Setup Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '18'
      - name: Install dependencies
        run: npm install html-validator axe-core
      - name: Validate HTML
        run: node scripts/html-validator.js
      - name: Accessibility check
        run: node scripts/accessibility-check.js
```
