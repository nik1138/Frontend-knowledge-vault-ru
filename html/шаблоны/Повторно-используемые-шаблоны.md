# Повторно используемые шаблоны

## Введение в повторно используемые шаблоны

Повторно используемые шаблоны - это модульные HTML-структуры, которые можно использовать в разных частях веб-сайта или в разных проектах. В 2025 году в российской веб-разработке особое внимание уделяется модульности, переиспользуемости компонентов и созданию библиотек шаблонов для ускорения разработки и обеспечения согласованности интерфейса.

```html
  <div class="input-container">
    <label class="input-label" :for="id">
      <slot name="label" />
    </label>
    <input 
      class="input-field" 
      :id="id"
      :type="type"
      :name="name"
      :placeholder="placeholder"
      :required="required"
      :aria-describedby="errorId"
    >
    <div class="input-hint">
      <slot name="hint" />
    </div>
    <div class="input-error" :id="errorId">
      <slot name="error" />
    </div>
  </div>
</template>

<script>
class InputFieldComponent extends HTMLElement {
  static get observedAttributes() {
    return ['type', 'name', 'placeholder', 'required', 'id'];
  }
  
  constructor() {
    super();
    
    const template = document.getElementById('input-field-template');
    const templateContent = template.content;
    
    this.attachShadow({mode: 'open'})
      .appendChild(templateContent.cloneNode(true));
  }
  
  attributeChangedCallback(name, oldValue, newValue) {
    if (this.shadowRoot) {
      const input = this.shadowRoot.querySelector('.input-field');
      if (input) {
        input.setAttribute(name, newValue);
      }
    }
  }
  
  connectedCallback() {
    // Установка атрибутов из свойств
    this.id = this.getAttribute('id') || 'input-' + Math.random().toString(36).substr(2, 9);
    this.type = this.getAttribute('type') || 'text';
    this.name = this.getAttribute('name') || '';
    this.placeholder = this.getAttribute('placeholder') || '';
    this.required = this.hasAttribute('required');
    
    // Установка ID для ошибки
    const errorElement = this.shadowRoot.querySelector('.input-error');
    if (errorElement) {
      errorElement.id = this.id + '-error';
    }
  }
}

customElements.define('input-field', InputFieldComponent);
</script>
```

```html
        <slot name="footer" />
      </div>
    </div>
  </div>
</template>

<script>
class ModalComponent extends HTMLElement {
  constructor() {
    super();
    
    const template = document.getElementById('modal-template');
    const templateContent = template.content;
    
    this.attachShadow({mode: 'open'})
      .appendChild(templateContent.cloneNode(true));
  }
  
  connectedCallback() {
    const modal = this.shadowRoot.querySelector('.modal');
    const closeBtn = this.shadowRoot.querySelector('.close-btn');
    
    // Обработчик закрытия по кнопке
    closeBtn.addEventListener('click', () => {
      this.close();
    });
    
    // Обработчик закрытия по клику вне модального окна
    modal.addEventListener('click', (e) => {
      if (e.target === modal) {
        this.close();
      }
    });
    
    // Обработчик клавиши Escape
    this.handleEscape = (e) => {
      if (e.key === 'Escape') {
        this.close();
      }
    };
    
    document.addEventListener('keydown', this.handleEscape);
  }
  
  disconnectedCallback() {
    document.removeEventListener('keydown', this.handleEscape);
  }
  
  open() {
    this.shadowRoot.querySelector('.modal').classList.add('active');
    document.body.style.overflow = 'hidden'; // Предотвращение прокрутки фона
  }
  
  close() {
    this.shadowRoot.querySelector('.modal').classList.remove('active');
    document.body.style.overflow = ''; // Восстановление прокрутки
  }
}

customElements.define('modal-component', ModalComponent);
</script>
```

```html
    }

      background-color: #e9ecef;
    }
  </style>
  
  <div class="list-container">
    <slot name="list-items" />
  </div>
</template>

<script>
class ListComponent extends HTMLElement {
  constructor() {
    super();
    
    const template = document.getElementById('list-template');
    const templateContent = template.content;
    
    this.attachShadow({mode: 'open'})
      .appendChild(templateContent.cloneNode(true));
  }
  
  addItem(itemData) {
    const item = document.createElement('div');
    item.className = 'list-item';
    item.innerHTML = `
      <span class="item-title">${itemData.title}</span>
      <div class="list-actions">
        <button class="btn edit-btn" data-id="${itemData.id}">Редактировать</button>
        <button class="btn delete-btn" data-id="${itemData.id}">Удалить</button>
      </div>
    `;
    
    // Добавляем обработчики событий
    const editBtn = item.querySelector('.edit-btn');
    const deleteBtn = item.querySelector('.delete-btn');
    
    if (editBtn) {
      editBtn.addEventListener('click', () => {
        this.dispatchEvent(new CustomEvent('item-edit', {
          detail: { id: itemData.id, title: itemData.title },
          bubbles: true
        }));
      });
    }
    
    if (deleteBtn) {
      deleteBtn.addEventListener('click', () => {
        this.dispatchEvent(new CustomEvent('item-delete', {
          detail: { id: itemData.id },
          bubbles: true
        }));
      });
    }
    
    this.shadowRoot.querySelector('.list-container').appendChild(item);
  }
  
  setItems(items) {
    // Очистка существующих элементов
    const container = this.shadowRoot.querySelector('.list-container');
    while (container.children.length > 1) { // Сохраняем шаблон
      container.removeChild(container.lastChild);
    }
    
    // Добавление новых элементов
    items.forEach(item => this.addItem(item));
  }
  
  selectItem(id) {
    // Снимаем выделение со всех элементов
    this.shadowRoot.querySelectorAll('.list-item').forEach(item => {
      item.classList.remove('selected');
    });
    
    // Выделяем указанный элемент
    const itemToSelect = Array.from(this.shadowRoot.querySelectorAll('.list-item'))
      .find(item => item.querySelector('.edit-btn').dataset.id === id);
      
    if (itemToSelect) {
      itemToSelect.classList.add('selected');
    }
  }
}

customElements.define('list-component', ListComponent);
</script>
```

## Практические рекомендации

1. **Модульность**: Создавайте компоненты, которые могут быть использованы в разных контекстах без изменений.

2. **Повторное использование**: Используйте одинаковые компоненты для схожих элементов интерфейса.

3. **Конфигурируемость**: Обеспечьте возможность настройки компонентов через атрибуты и слоты.

4. **Документация**: Создавайте документацию для каждого компонента с примерами использования.

5. **Тестирование**: Проводите тестирование компонентов в разных браузерах и на разных устройствах.

6. **Совместимость**: Убедитесь, что компоненты работают с полифиллами для старых браузеров.

7. **Производительность**: Оптимизируйте компоненты для быстрой загрузки и работы.

8. **Доступность**: Обеспечьте доступность всех компонентов для пользователей с ограниченными возможностями.
