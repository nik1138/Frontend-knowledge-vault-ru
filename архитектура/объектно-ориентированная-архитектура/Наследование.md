class CompositeValidator extends Validator {
  constructor(validators = []) {
    super();
    this.validators = validators;
  }

  addValidator(validator) {
    if (validator instanceof Validator) {
      this.validators.push(validator);
    }
  }

  validate(value) {
    return this.validators.every(validator => validator.validate(value));
  }

  getErrorMessage() {
    const messages = this.validators
      .filter(validator => !validator.validate(value))
      .map(validator => validator.getErrorMessage());
    return messages.join('; ');
  }

  getErrorMessages(value) {
    return this.validators
      .filter(validator => !validator.validate(value))
      .map(validator => validator.getErrorMessage());
  }

class BaseForm {
  constructor(container, options = {}) {
    this.container = container;
    this.options = options;
    this.fields = new Map();
    this.validators = new Map();
    this.onSubmit = options.onSubmit || (() => {});
  }

  addField(name, field) {
    this.fields.set(name, field);
    field.on('change', (value) => {
      this.validateField(name);
    });
  }

  addValidator(fieldName, validator) {
    if (!this.validators.has(fieldName)) {
      this.validators.set(fieldName, []);
    }
    this.validators.get(fieldName).push(validator);
  }

  validateField(fieldName) {
    const field = this.fields.get(fieldName);
    const validators = this.validators.get(fieldName) || [];
    
    if (!field || !validators.length) {
      return true;
    }

    const value = field.getValue();
    const errors = [];

    for (const validator of validators) {
      if (!validator.validate(value)) {
        errors.push(validator.getErrorMessage());

    field.setErrors(errors);
    return errors.length === 0;

    let isValid = true;
    for (const fieldName of this.fields.keys()) {

        isValid = false;

    return isValid;

abstract class BaseComponent<TProps = {}, TState = {}> {

  protected state: TState;
  protected element: HTMLElement | null = null;

  constructor(props: TProps) {
    this.props = props;
    this.state = this.getDefaultState();
    this.init();
  }

  protected abstract getDefaultState(): TState;
  protected abstract render(): HTMLElement;

    this.element = this.render();
  }

  public updateState(newState: Partial<TState>): void {
    this.state = { ...this.state, ...newState };
    if (this.element) {
      const newElement = this.render();
      this.element.parentNode?.replaceChild(newElement, this.element);
      this.element = newElement;
    }
  }
}

  onClick: () => void;
  disabled?: boolean;
}

class ButtonComponent extends BaseComponent<ButtonProps, { clicked: boolean }> {

    return { clicked: false };
  }

    const button = document.createElement('button');
    button.textContent = this.props.text;
    button.disabled = this.props.disabled || false;

      this.props.onClick();

  }
