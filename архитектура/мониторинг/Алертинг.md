class AlertAggregator {

    this.aggregationWindow = aggregationWindow;
    this.pendingAlerts = new Map();
  }

  process(alert) {
    const key = this.generateAggregationKey(alert);
    const now = Date.now();
    
    if (!this.pendingAlerts.has(key)) {
      this.pendingAlerts.set(key, {
        alerts: [],
        firstOccurrence: now
      });
    }
    
    const group = this.pendingAlerts.get(key);
    group.alerts.push(alert);
    
    // –ï—Å–ª–∏ –ø—Ä–æ—à–ª–æ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –≤—Ä–µ–º–µ–Ω–∏, –æ—Ç–ø—Ä–∞–≤–ª—è–µ–º –∞–≥—Ä–µ–≥–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –∞–ª–µ—Ä—Ç
    if (now - group.firstOccurrence >= this.aggregationWindow) {
      return this.sendAggregatedAlert(group);
    }
    
    return null;
  }

  generateAggregationKey(alert) {
    return `${alert.name}-${Math.floor(alert.timestamp / this.aggregationWindow)}`;
  }

  sendAggregatedAlert(group) {
    const aggregatedAlert = {
      id: `aggregated-${Date.now()}`,
      name: `–ê–≥—Ä–µ–≥–∏—Ä–æ–≤–∞–Ω–Ω—ã–π: ${group.alerts[0].name}`,

      count: group.alerts.length,
      timestamp: group.firstOccurrence
    };
    
    // –û—á–∏—â–∞–µ–º –≥—Ä—É–ø–ø—É
    this.pendingAlerts.delete(this.generateAggregationKey(group.alerts[0]));
    
    return aggregatedAlert;
  }

  getHighestSeverity(alerts) {
    const severityOrder = ['low', 'medium', 'high', 'critical'];
    return alerts.reduce((highest, alert) => {
      const currentIdx = severityOrder.indexOf(alert.severity);
      const highestIdx = severityOrder.indexOf(highest);
      return currentIdx > highestIdx ? alert.severity : highest;
    }, 'low');
  }
