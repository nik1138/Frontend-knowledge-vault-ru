## Пример: Бинарный поиск с модификациями

Для решения задач на уровне Middle часто требуется модификация стандартных алгоритмов под конкретные условия.

### Псевдокод:
```
ФУНКЦИЯ бинарный_поиск_с_условием(массив, цель):
    лево = 0
    право = длина(массив) - 1
    
    ПОКА лево <= право:
        середина = (лево + право) / 2
        
        ЕСЛИ массив[середина] == цель:
            ВОЗВРАТ середина
        ИНАЧЕ ЕСЛИ условие_специальное(массив[середина], цель):
            право = середина - 1
        ИНАЧЕ:
            лево = середина + 1
    
    ВОЗВРАТ -1
```

```javascript
function binarySearchWithCondition(arr, target, conditionFn) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        const midValue = arr[mid];
        
        if (midValue === target) {
            return mid;
        } else if (conditionFn(midValue, target)) {
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    return -1;
}

// Пример использования: поиск элемента в частично отсортированном массиве
function findInRotatedArray(nums, target) {
    let left = 0;
    let right = nums.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        
        if (nums[mid] === target) {
            return mid;
        }
        
        // Проверяем, какая половина отсортирована
        if (nums[left] <= nums[mid]) {
            // Левая половина отсортирована
            if (nums[left] <= target && target < nums[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else {
            // Правая половина отсортирована
            if (nums[mid] < target && target <= nums[right]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    
    return -1;
}
```
