Пример алгоритма из книги - алгоритм поиска подстроки (алгоритм Кнута-Морриса-Пратта):

**Псевдокод:**
```
ФУНКЦИЯ КМП_поиск(текст, шаблон)
    НАЧАЛО
        длина_шаблона = длина(шаблон)
        длина_текста = длина(текст)

        ЕСЛИ длина_шаблона = 0
            ВОЗВРАТ 0
        КОНЕЦ ЕСЛИ

        префикс_функция = вычислить_префикс_функцию(шаблон)
        j = 0

        ДЛЯ i = 0 ДО длина_текста - 1
            ПОКА j > 0 И текст[i] ≠ шаблон[j]
                j = префикс_функция[j - 1]
            КОНЕЦ ПОКА

            ЕСЛИ текст[i] = шаблон[j]
                j = j + 1

            ЕСЛИ j = длина_шаблона
                ВОЗВРАТ i - длина_шаблона + 1
            КОНЕЦ ЕСЛИ
        КОНЕЦ ДЛЯ

        ВОЗВРАТ -1  // не найдено
    КОНЕЦ

ФУНКЦИЯ вычислить_префикс_функцию(шаблон)
    НАЧАЛО
        длина = длина(шаблон)
        префикс = массив(длина)
        префикс[0] = 0
        длина_префикса = 0
        i = 1

        ПОКА i < длина
            ЕСЛИ шаблон[i] = шаблон[длина_префикса]
                длина_префикса = длина_префикса + 1
                префикс[i] = длина_префикса
                i = i + 1
            ИНАЧЕ
                ЕСЛИ длина_префикса ≠ 0
                    длина_префикса = префикс[длина_префикса - 1]
                ИНАЧЕ
                    префикс[i] = 0
                    i = i + 1
                КОНЕЦ ЕСЛИ
            КОНЕЦ ЕСЛИ
        КОНЕЦ ПОКА

        ВОЗВРАТ префикс
    КОНЕЦ
```

**JavaScript реализация:**
```javascript
function kmpSearch(text, pattern) {
    if (pattern.length === 0) return 0;

    const lps = computeLPSArray(pattern);
    let i = 0; // индекс для text
    let j = 0; // индекс для pattern

    while (i < text.length) {
        if (pattern[j] === text[i]) {
            i++;
            j++;
        }

        if (j === pattern.length) {
            return i - j; // найдено совпадение
        } else if (i < text.length && pattern[j] !== text[i]) {
            if (j !== 0) {
                j = lps[j - 1];
            } else {
                i++;
            }
        }
    }

    return -1; // не найдено
}

function computeLPSArray(pattern) {
    const lps = new Array(pattern.length);
    let len = 0;
    let i = 1;

    lps[0] = 0;

    while (i < pattern.length) {
        if (pattern[i] === pattern[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len !== 0) {
                len = lps[len - 1];
            } else {
                lps[i] = 0;
                i++;
            }
        }
    }

    return lps;
}
```
