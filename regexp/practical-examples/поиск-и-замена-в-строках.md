function replacePreservingCase(text, search, replacement) {
  return text.replace(new RegExp(search, 'gi'), function(match) {
    // –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–µ–≥–∏—Å—Ç—Ä –ø–µ—Ä–≤–æ–≥–æ —Å–∏–º–≤–æ–ª–∞

      return replacement.toLowerCase();
    } else if (match[0] === match[0].toUpperCase()) {
      return replacement.charAt(0).toUpperCase() + replacement.slice(1).toLowerCase();
    }
    return replacement;
  });
}

  // –ü–æ–∏—Å–∫ –∏ –ø–æ–¥—Å–≤–µ—Ç–∫–∞ —Ç–µ–∫—Å—Ç–∞
  static highlightText(text, searchTerm, caseSensitive = false) {
    if (!searchTerm) return text;
    
    // –≠–∫—Ä–∞–Ω–∏—Ä—É–µ–º —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã –≤ –ø–æ–∏—Å–∫–æ–≤–æ–º –∑–∞–ø—Ä–æ—Å–µ
    const escapedTerm = searchTerm.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');

    const flags = caseSensitive ? 'g' : 'gi';
    const regex = new RegExp(`(${escapedTerm})`, flags);
    
    // –ó–∞–º–µ–Ω—è–µ–º —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è —Å –ø–æ–¥—Å–≤–µ—Ç–∫–æ–π

  static findWithPositions(text, searchTerm, caseSensitive = false) {
    if (!searchTerm) return [];
    
    const escapedTerm = searchTerm.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const flags = caseSensitive ? 'g' : 'gi';
    const regex = new RegExp(`(${escapedTerm})`, flags);
    
    const results = [];
    let match;
    
    while ((match = regex.exec(text)) !== null) {
      results.push({
        match: match[0],
        index: match.index,
        endIndex: match.index + match[0].length
      });
    }
    
    return results;
  }
  
  // –ó–∞–º–µ–Ω–∞ —Ç–æ–ª—å–∫–æ –ø–µ—Ä–≤–æ–≥–æ –≤—Ö–æ–∂–¥–µ–Ω–∏—è
  static replaceFirst(text, searchTerm, replacement) {
    const escapedTerm = searchTerm.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(escapedTerm);
    return text.replace(regex, replacement);
  }
  
  // –ó–∞–º–µ–Ω–∞ —Å –ø–æ–¥—Å—á–µ—Ç–æ–º –∏–∑–º–µ–Ω–µ–Ω–∏–π
  static replaceAndCount(text, searchTerm, replacement) {
    const escapedTerm = searchTerm.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(escapedTerm, 'g');
    
    let count = 0;
    const result = text.replace(regex, (match) => {
      count++;
      return replacement;
    });
    
    return { text: result, count };
  }
  
  // –ú–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω–∞—è –∑–∞–º–µ–Ω–∞

      result = result.replace(regex, replacement);
    }
    
    return result;
  }
}
