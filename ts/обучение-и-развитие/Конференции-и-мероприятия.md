## Темы докладов

### Популярные темы на TypeScript конференциях

```typescript
// Пример кода, который может быть представлен на конференции
type ConferenceTopic = 'Advanced Types' | 'Generics' | 'Decorators' | 'Type Safety' | 'Performance';

interface ConferenceTalk {
  title: string;
  speaker: string;
  duration: number; // в минутах
  topics: ConferenceTopic[];
  level: 'beginner' | 'intermediate' | 'advanced';
}

class ConferenceSchedule {
  private talks: ConferenceTalk[] = [];
  
  addTalk(talk: ConferenceTalk): void {
    this.talks.push(talk);
    console.log(`Добавлен доклад: ${talk.title}`);
  }
  
  getTalksByTopic(topic: ConferenceTopic): ConferenceTalk[] {
    return this.talks.filter(talk => talk.topics.includes(topic));
  }
  
  getTalksByLevel(level: 'beginner' | 'intermediate' | 'advanced'): ConferenceTalk[] {
    return this.talks.filter(talk => talk.level === level);
  }
}

// Использование
const schedule = new ConferenceSchedule();
schedule.addTalk({
  title: 'Глубокое погружение в дженерики',
  speaker: 'Иван Петров',
  duration: 45,
  topics: ['Generics', 'Advanced Types'],
  level: 'advanced'
});
```
