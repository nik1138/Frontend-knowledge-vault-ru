# Решения распространенных задач

## Введение

В этом разделе мы рассмотрим решения наиболее распространенных задач, с которыми сталкиваются разработчики при использовании TypeScript в frontend разработке. Эти примеры помогут ускорить разработку и избежать типичных ошибок.

## Работа с асинхронными операциями

### Типизация асинхронных функций

```ts
// Типизация асинхронных функций
interface User {
  id: number;
  name: string;
  email: string;
}

// Базовая асинхронная функция с типизацией
async function fetchUser(id: number): Promise<User> {
  const response = await fetch(`/api/users/${id}`);
  if (!response.ok) {
    throw new Error(`Failed to fetch user with id ${id}`);
  }
  return response.json();
}

// Асинхронная функция с обработкой ошибок
async function fetchUserSafe(id: number): Promise<User | null> {
  try {
    return await fetchUser(id);
  } catch (error) {
    console.error('Error fetching user:', error);
    return null;
  }
}

// Параллельное выполнение асинхронных операций
async function fetchMultipleUsers(ids: number[]): Promise<User[]> {
  const promises = ids.map(id => fetchUser(id));
  const results = await Promise.all(promises);
  return results.filter(user => user !== null) as User[];
}

// Последовательное выполнение асинхронных операций
async function processUsersSequentially(ids: number[]): Promise<User[]> {
  const users: User[] = [];
  for (const id of ids) {
    try {
      const user = await fetchUser(id);
      users.push(user);
    } catch (error) {
      console.error(`Error fetching user with id ${id}:`, error);
    }
  }
  return users;
}
```

### Типизация для обработки ошибок

```ts
// Типизированный обработчик ошибок
interface ApiError {
  message: string;
  status: number;
  code?: string;
}

function isApiError(error: unknown): error is ApiError {
  return (
    typeof error === 'object' &&
    error !== null &&
    'message' in error &&
    'status' in error
  );
}

// Обработка ошибок с типизацией
async function fetchWithErrorHandling<T>(url: string): Promise<T> {
  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      const errorData: ApiError = await response.json().catch(() => ({
        message: 'Unknown error',
        status: response.status
      }));
      
      throw errorData;
    }
    
    return await response.json();
  } catch (error) {
    if (isApiError(error)) {
      console.error(`API Error: ${error.message} (Status: ${error.status})`);
      throw error;
    } else if (error instanceof TypeError) {
      console.error('Network error:', error.message);
      throw new Error('Network connection failed');
    } else {
      console.error('Unexpected error:', error);
      throw new Error('An unexpected error occurred');
    }
  }
}
```

## Работа с формами и валидацией

### Типизация форм

```ts
// Интерфейсы для формы регистрации
interface RegistrationFormValues {
  firstName: string;
  lastName: string;
  email: string;
  password: string;
  confirmPassword: string;
  age: number;
  agreeToTerms: boolean;
}

interface FormErrors {
  firstName?: string;
  lastName?: string;
  email?: string;
  password?: string;
  confirmPassword?: string;
  age?: string;
  agreeToTerms?: string;
}

// Валидация формы
function validateRegistrationForm(values: RegistrationFormValues): FormErrors {
  const errors: FormErrors = {};

  if (!values.firstName.trim()) {
    errors.firstName = 'First name is required';
  } else if (values.firstName.length < 2) {
    errors.firstName = 'First name must be at least 2 characters';
  }

  if (!values.lastName.trim()) {
    errors.lastName = 'Last name is required';
  }

  if (!values.email) {
    errors.email = 'Email is required';
  } else if (!/\S+@\S+\.\S+/.test(values.email)) {
    errors.email = 'Email is invalid';
  }

  if (!values.password) {
    errors.password = 'Password is required';
  } else if (values.password.length < 8) {
    errors.password = 'Password must be at least 8 characters';
  } else if (!/(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/.test(values.password)) {
    errors.password = 'Password must contain at least one uppercase letter, one lowercase letter, and one number';
  }

  if (values.password !== values.confirmPassword) {
    errors.confirmPassword = 'Passwords do not match';
  }

  if (values.age < 13) {
    errors.age = 'You must be at least 13 years old';
  }

  if (!values.agreeToTerms) {
    errors.agreeToTerms = 'You must agree to the terms';
  }

  return errors;
}

// Хук для управления формой
import { useState, useCallback } from 'react';

function useForm<T>(
  initialValues: T,
  validate: (values: T) => Record<string, string>
) {
  const [values, setValues] = useState<T>(initialValues);
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleChange = useCallback((name: keyof T, value: any) => {
    setValues(prev => ({ ...prev, [name]: value }));
    
    // Очистка ошибки при изменении поля
    setErrors(prev => {
      const newErrors = { ...prev };
      delete newErrors[name as string];
      return newErrors;
    });
  }, []);

  const handleSubmit = useCallback(async (onSubmit: (values: T) => Promise<void>) => {
    const validationErrors = validate(values);
    setErrors(validationErrors);

    if (Object.keys(validationErrors).length === 0) {
      setIsSubmitting(true);
      try {
        await onSubmit(values);
      } catch (error) {
        console.error('Form submission error:', error);
      } finally {
        setIsSubmitting(false);
      }
    }
  }, [values, validate]);

  return {
    values,
    errors,
    isSubmitting,
    handleChange,
    handleSubmit
  };
}

// Использование хука
const useRegistrationForm = () => useForm<RegistrationFormValues>(
  { 
    firstName: '', 
    lastName: '', 
    email: '', 
    password: '', 
    confirmPassword: '', 
    age: 18, 
    agreeToTerms: false 
  },
  validateRegistrationForm
);
```

## Работа с localStorage и сессией

### Типизированный сервис для хранения данных

```ts
// Типизированный сервис для работы с localStorage
class StorageService {
  // Общий метод для получения данных с типизацией
  static getItem<T>(key: string, defaultValue: T): T {
    try {
      const item = localStorage.getItem(key);
      return item ? JSON.parse(item) : defaultValue;
    } catch (error) {
      console.error(`Error parsing localStorage item with key "${key}":`, error);
      return defaultValue;
    }
  }

  // Общий метод для сохранения данных
  static setItem<T>(key: string, value: T): void {
    try {
      localStorage.setItem(key, JSON.stringify(value));
    } catch (error) {
      console.error(`Error storing item with key "${key}":`, error);
    }
  }

  // Удаление элемента
  static removeItem(key: string): void {
    try {
      localStorage.removeItem(key);
    } catch (error) {
      console.error(`Error removing item with key "${key}":`, error);
    }
  }

  // Очистка всех данных
  static clear(): void {
    try {
      localStorage.clear();
    } catch (error) {
      console.error('Error clearing localStorage:', error);
    }
  }
}

// Типизированный хук для localStorage
import { useState, useEffect } from 'react';

function useLocalStorage<T>(key: string, initialValue: T): [T, (value: T) => void] {
  const [storedValue, setStoredValue] = useState<T>(() => {
    return StorageService.getItem<T>(key, initialValue);
  });

  const setValue = (value: T) => {
    try {
      setStoredValue(value);
      StorageService.setItem<T>(key, value);
    } catch (error) {
      console.error(error);
    }
  };

  return [storedValue, setValue];
}

// Использование для пользовательских настроек
interface UserPreferences {
  theme: 'light' | 'dark';
  language: 'en' | 'ru' | 'es';
  notifications: boolean;
  fontSize: 'small' | 'medium' | 'large';
}

const [preferences, setPreferences] = useLocalStorage<UserPreferences>(
  'user-preferences',
  { 
    theme: 'light', 
    language: 'en', 
    notifications: true, 
    fontSize: 'medium' 
  }
);
```

```tsx
import React, { useState } from 'react';

// Типизация обработчиков событий
interface FormState {
  name: string;
  email: string;
  message: string;
}

const ContactForm: React.FC = () => {
  const [formState, setFormState] = useState<FormState>({
    name: '',
    email: '',
    message: ''
  });

  // Типизация обработчика изменения инпута
  const handleInputChange = (
    e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
  ) => {
    const { name, value } = e.target;
    setFormState(prev => ({
      ...prev,
      [name]: value
    }));
  };

  // Типизация обработчика отправки формы
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    console.log('Form submitted:', formState);
    // Отправка данных
  };

  // Типизация обработчика клика
  const handleReset = (e: React.MouseEvent<HTMLButtonElement>) => {
    e.preventDefault();
    setFormState({
      name: '',
      email: '',
      message: ''
    });
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="name">Name:</label>
        <input
          type="text"
          id="name"
          name="name"
          value={formState.name}
          onChange={handleInputChange}
        />
      </div>
      
      <div>
        <label htmlFor="email">Email:</label>
        <input
          type="email"
          id="email"
          name="email"
          value={formState.email}
          onChange={handleInputChange}
        />
      </div>
      
      <div>
        <label htmlFor="message">Message:</label>
        <textarea
          id="message"
          name="message"
          value={formState.message}
          onChange={handleInputChange}
        />
      </div>
      
      <button type="submit">Send</button>
      <button type="button" onClick={handleReset}>Reset</button>
    </form>
  );
};
```

### Утилиты для работы с массивами

```ts
// Типизированные утилиты для работы с массивами
interface User {
  id: number;
  name: string;
  email: string;
  department: string;
}

// Группировка элементов по свойству
function groupBy<T, K extends keyof any>(array: T[], key: (item: T) => K): Record<K, T[]> {
  return array.reduce((result, item) => {
    const groupKey = key(item);
    if (!result[groupKey]) {
      result[groupKey] = [];
    }
    result[groupKey].push(item);
    return result;
  }, {} as Record<K, T[]>);
}

// Пример использования
const users: User[] = [
  { id: 1, name: 'John', email: 'john@example.com', department: 'IT' },
  { id: 2, name: 'Jane', email: 'jane@example.com', department: 'HR' },
  { id: 3, name: 'Bob', email: 'bob@example.com', department: 'IT' },
];

const usersByDepartment = groupBy(users, user => user.department);
// Результат: { IT: [user1, user3], HR: [user2] }

// Поиск уникальных элементов
function uniqueBy<T, K>(array: T[], keyFn: (item: T) => K): T[] {
  const seen = new Set<K>();
  return array.filter(item => {
    const key = keyFn(item);
    if (seen.has(key)) {
      return false;
    }
    seen.add(key);
    return true;
  });
}

// Сортировка с типизацией
function sortBy<T>(array: T[], keyFn: (item: T) => any, order: 'asc' | 'desc' = 'asc'): T[] {
  return [...array].sort((a, b) => {
    const aVal = keyFn(a);
    const bVal = keyFn(b);
    
    if (aVal < bVal) {
      return order === 'asc' ? -1 : 1;
    }
    if (aVal > bVal) {
      return order === 'asc' ? 1 : -1;
    }
    return 0;
  });
}

// Пример сортировки
const sortedUsers = sortBy(users, user => user.name, 'asc');
```

```ts
// Глубокое объединение объектов
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

function deepMerge<T extends Record<string, any>>(target: T, source: DeepPartial<T>): T {
  const result: any = { ...target };
  
  for (const key in source) {
    if (source.hasOwnProperty(key)) {
      if (
        typeof source[key] === 'object' &&
        source[key] !== null &&
        !Array.isArray(source[key]) &&
        typeof target[key] === 'object' &&
        target[key] !== null
      ) {
        result[key] = deepMerge(target[key], source[key]);
      } else {
        result[key] = source[key];
      }
    }
  }
  
  return result;
}

// Пример использования
interface Config {
  api: {
    baseUrl: string;
    timeout: number;
  };
  ui: {
    theme: string;
    language: string;
  };
}

const defaultConfig: Config = {
  api: {
    baseUrl: 'https://api.example.com',
    timeout: 5000
  },
  ui: {
    theme: 'light',
    language: 'en'
  }
};

const userConfig: DeepPartial<Config> = {
  api: {
    baseUrl: 'https://my-api.com'
  },
  ui: {
    theme: 'dark'
  }
};

const finalConfig = deepMerge(defaultConfig, userConfig);
```

```ts
  }));

const mockProducts = generateMockProducts(10);
```

## Практические примеры использования

### Типизация для кастомных хуков

```tsx
import { useState, useEffect, useRef } from 'react';

// Хук для отслеживания изменений
function usePrevious<T>(value: T): T | undefined {
  const ref = useRef<T>();
  
  useEffect(() => {
    ref.current = value;
  }, [value]);
  
  return ref.current;
}

// Хук для дебаунса
function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}

// Хук для работы с API с автоматической отменой запросов
function useApi<T>(url: string, options?: RequestInit) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);
  const cancelRef = useRef(false);

  useEffect(() => {
    cancelRef.current = false;
    
    const fetchData = async () => {
      try {
        setLoading(true);
        setError(null);
        
        const response = await fetch(url, options);
        
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const result: T = await response.json();
        
        if (!cancelRef.current) {
          setData(result);
        }
      } catch (err) {
        if (!cancelRef.current) {
          setError(err instanceof Error ? err.message : 'An error occurred');
        }
      } finally {
        if (!cancelRef.current) {
          setLoading(false);
        }
      }
    };

    fetchData();

    return () => {
      cancelRef.current = true;
    };
  }, [url]);

  return { data, loading, error };
}
```

### Типизация для компонентов с динамическими пропсами

```tsx
import React from 'react';

// Утилита для создания компонентов с поддержкой полиморфизма
type PolymorphicComponent<Props, DefaultElement extends React.ElementType> = {
  <Element extends React.ElementType = DefaultElement>(
    props: {
      as?: Element;
    } & React.ComponentPropsWithoutRef<Element> & 
    Omit<Props, keyof React.ComponentPropsWithoutRef<Element>>
  ): React.ReactElement | null;
} & Omit<React.NamedExoticComponent<Props>, 'propTypes' | 'contextTypes' | 'displayName'>;

interface BoxOwnProps {
  children: React.ReactNode;
  className?: string;
  padding?: number;
  margin?: number;
}

type BoxProps<T extends React.ElementType> = BoxOwnProps & {
  as?: T;
};

const Box = React.forwardRef(
  <T extends React.ElementType = 'div'>(
    { as, children, className = '', padding = 0, margin = 0, ...props }: BoxProps<T>,
    ref: React.Ref<HTMLElement>
  ) => {
    const Component = as || 'div';
    
    const style = {
      padding: `${padding}px`,
      margin: `${margin}px`,
    };
    
    return (
      <Component
        ref={ref}
        className={className}
        style={style}
        {...props}
      >
        {children}
      </Component>
    );
  }
) as PolymorphicComponent<BoxOwnProps, 'div'>;

// Использование полиморфного компонента
const Example = () => (
  <div>
    <Box as="section" padding={20} margin={10}>
      This is a section
    </Box>
    <Box as="article" padding={15}>
      This is an article
    </Box>
    <Box padding={10} className="custom-class">
      This is a div (default)
    </Box>
  </div>
);
```
