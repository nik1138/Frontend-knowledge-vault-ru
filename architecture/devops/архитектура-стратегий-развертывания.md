// blue-green-deployment-script.js
class BlueGreenDeployment {
  constructor(k8sClient, config) {
    this.k8s = k8sClient;
    this.config = config;
    this.currentColor = this.getCurrentColor();
  }
  
  async deployNewVersion(image) {
    const newColor = this.currentColor === 'blue' ? 'green' : 'blue';
    
    // 1. –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –Ω–æ–≤–æ–≥–æ Deployment
    await this.updateDeployment(newColor, image);
    
    // 2. –ó–∞–ø—É—Å–∫ –∞–Ω–∞–ª–∏–∑–∞ —Ä–∞–±–æ—Ç–æ—Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏
    const analysis = await this.runAnalysis(newColor);
    
    if (analysis.success) {
      // 3. –ü–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ —Ç—Ä–∞—Ñ–∏–∫–∞

      // 4. –ü–æ–º–µ—Ç–∫–∞ —Å—Ç–∞—Ä–æ–π –≤–µ—Ä—Å–∏–∏ –¥–ª—è —É–¥–∞–ª–µ–Ω–∏—è
      await this.markOldDeploymentForCleanup(this.currentColor);
      
      console.log(`Successfully switched traffic to ${newColor} version`);
      return { success: true, newColor };
    } else {
      console.error('Analysis failed, keeping current version');
      return { success: false, reason: 'Health check failed' };
    }
  }
  
  async runAnalysis(color) {
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–æ—Å—Ç—É–ø–Ω–æ—Å—Ç–∏ –∏ —Ä–∞–±–æ—Ç–æ—Å–ø–æ—Å–æ–±–Ω–æ—Å—Ç–∏ —Å–µ—Ä–≤–∏—Å–∞
    const healthCheckUrl = `http://${this.config.previewServiceName}.${this.config.namespace}`;
    
    try {
      // –ó–∞–ø—É—Å–∫ —Å–µ—Ä–∏–∏ –ø—Ä–æ–≤–µ—Ä–æ–∫
      let checksPassed = 0;
      for (let i = 0; i < this.config.healthChecks.total; i++) {
        if (await this.performHealthCheck(healthCheckUrl)) {
          checksPassed++;
        }
        
        if (checksPassed >= this.config.healthChecks.minimum) {
          return { success: true, passed: checksPassed };
        }
        
        await this.sleep(this.config.healthChecks.interval);
      }
      
      return { success: checksPassed >= this.config.healthChecks.minimum, passed: checksPassed };
    } catch (error) {
      console.error('Analysis error:', error);
      return { success: false, error };
    }
  }
  
  async performHealthCheck(url) {
    try {
      const response = await fetch(`${url}/health`, { timeout: 5000 });
      return response.ok && response.status === 200;
    } catch (error) {
      return false;
    }
  }
  
  switchTraffic(targetColor) {
    // –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –æ—Å–Ω–æ–≤–Ω–æ–≥–æ —Å–µ—Ä–≤–∏—Å–∞ –¥–ª—è —É–∫–∞–∑–∞–Ω–∏—è –Ω–∞ –Ω–æ–≤—É—é –≤–µ—Ä—Å–∏—é
    const servicePatch = {
      spec: {
        selector: {
          version: targetColor
        }
      }
    };
    
    return this.k8s.appsV1Api.patchNamespacedDeployment(
      'frontend-app-main-service',
      this.config.namespace,
      servicePatch,
      undefined,
      undefined,
      undefined,
      undefined,
      {
        headers: {
          'Content-Type': 'application/merge-patch+json'
        }
      }
    );
  }
}

// config-migration.js
class ConfigMigration {
  constructor(configStore) {
    this.configStore = configStore;
  }
  
  async migrateConfiguration(newVersion, configTransformations) {
    const currentConfig = await this.configStore.getConfiguration();
    const transformedConfig = { ...currentConfig };
    
    for (const [version, transformations] of Object.entries(configTransformations)) {
      if (this.versionCompare(version, newVersion) <= 0) {
        for (const transform of transformations) {
          transformedConfig[transform.key] = transform.func(transformedConfig[transform.key]);
        }
      }
    }
    
    // –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –Ω–æ–≤—É—é –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—é

    await this.configStore.addToHistory(newVersion, transformedConfig);
    
    return transformedConfig;
  }
  
  versionCompare(v1, v2) {
    // –°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Å–µ–º–∞–Ω—Ç–∏—á–µ—Å–∫–∏—Ö –≤–µ—Ä—Å–∏–π
    const parts1 = v1.split('.').map(Number);
    const parts2 = v2.split('.').map(Number);
    
    for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
      const part1 = parts1[i] || 0;
      const part2 = parts2[i] || 0;
      
      if (part1 > part2) return 1;
      if (part1 < part2) return -1;
    }
    
    return 0;
  }
}
