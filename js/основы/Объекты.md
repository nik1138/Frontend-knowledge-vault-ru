# Объекты в JavaScript

Объекты в JavaScript - это коллекции пар "ключ-значение", где ключи - это строки (или символы), а значения могут быть любого типа данных, включая другие объекты и функции. Объекты являются основой объектно-ориентированного программирования в JavaScript и одной из его самых важных концепций.

## Создание объектов

### Литерал объекта (рекомендуемый способ)

```javascript
const person = {
    name: "Александр",
    age: 35,
    city: "Москва",
    isDeveloper: true,
    greet: function() {
        return `Привет, меня зовут ${this.name}`;
    }
};

// Современный синтаксис методов
const personModern = {
    name: "Мария",
    age: 28,
    city: "Санкт-Петербург",
    greet() {  // сокращенный синтаксис метода
        return `Привет, меня зовут ${this.name}`;
    },
    introduce(greeting = "Привет") {
        return `${greeting}, меня зовут ${this.name}, мне ${this.age} лет`;
    }
};
```

### Конструктор объекта

```javascript
const car = new Object();
car.brand = "Toyota";
car.model = "Camry";
car.year = 2023;
car.getInfo = function() {
    return `${this.year} ${this.brand} ${this.model}`;
};
```

### Функция-конструктор

```javascript
function Person(name, age, city) {
    this.name = name;
    this.age = age;
    this.city = city;
    this.greet = function() {
        return `Привет, меня зовут ${this.name}`;
    };
}

const person1 = new Person("Иван", 30, "Новосибирск");
const person2 = new Person("Ольга", 25, "Екатеринбург");
```

### Метод Object.create()

```javascript
const animal = {
    eats: true,
    walk() {
        return "Животное идет";
    }
};

const rabbit = Object.create(animal);
rabbit.jumps = true;
rabbit.color = "белый";

console.log(rabbit.eats); // true (наследуется от animal)
console.log(rabbit.walk()); // "Животное идет" (наследуется от animal)
console.log(rabbit.jumps); // true (собственное свойство)
```

## Доступ к свойствам

### Через точку (для известных свойств)

```javascript
const user = {
    name: "Дмитрий",
    age: 32,
    "full name": "Дмитрий Петров"  // свойство с пробелом в названии
};

console.log(user.name); // "Дмитрий"
console.log(user.age);  // 32
```

### Через квадратные скобки (для динамических свойств)

```javascript
const propertyName = "name";
console.log(user[propertyName]); // "Дмитрий"
console.log(user["full name"]);  // "Дмитрий Петров" (для свойств с пробелами)

// Динамический доступ
const field = "age";
console.log(user[field]); // 32
```

## Изменение и добавление свойств

```javascript
const book = {
    title: "Война и мир",
    author: "Лев Толстой"
};

// Добавление нового свойства
book.year = 1869;
book.pages = 1225;

// Изменение существующего свойства
book.title = "Анна Каренина";
book.author = "Лев Толстой";

// Удаление свойства
delete book.pages;

console.log(book); // {title: "Анна Каренина", author: "Лев Толстой", year: 1869}
```

## Вычисляемые свойства (computed properties)

```javascript
const prefix = "user";
const id = 123;

const user = {
    [prefix + "Id"]: id,
    [prefix + "Name"]: "Анна",
    [`${prefix}Age`]: 28
};

console.log(user); // {userId: 123, userName: "Анна", userAge: 28}
```

## Методы объекта

```javascript
const calculator = {
    value: 0,
    
    add(num) {
        this.value += num;
        return this; // для цепочки вызовов
    },
    
    subtract(num) {
        this.value -= num;
        return this;
    },
    
    multiply(num) {
        this.value *= num;
        return this;
    },
    
    getResult() {
        return this.value;
    },
    
    reset() {
        this.value = 0;
        return this;
    }
};

// Цепочка вызовов
const result = calculator
    .add(10)
    .multiply(2)
    .subtract(5)
    .getResult();

console.log(result); // 15
```

## Свойства-геттеры и сеттеры

```javascript
const temperature = {
    _celsius: 0,  // внутреннее свойство (по соглашению с подчеркиванием)
    
    get fahrenheit() {
        return (this._celsius * 9/5) + 32;
    },
    
    set fahrenheit(value) {
        this._celsius = (value - 32) * 5/9;
    },
    
    get celsius() {
        return this._celsius;
    },
    
    set celsius(value) {
        this._celsius = value;
    }
};

temperature.celsius = 25;
console.log(temperature.fahrenheit); // 77

temperature.fahrenheit = 86;
console.log(temperature.celsius); // 30
```

## Основные методы объектов

### Object.keys(), Object.values(), Object.entries()

```javascript
const laptop = {
    brand: "Apple",
    model: "MacBook Pro",
    year: 2024,
    price: 150000
};

console.log(Object.keys(laptop));    // ["brand", "model", "year", "price"]
console.log(Object.values(laptop));  // ["Apple", "MacBook Pro", 2024, 150000]
console.log(Object.entries(laptop)); // [["brand", "Apple"], ["model", "MacBook Pro"], ...]
```

### Object.assign()

```javascript
const defaults = {
    theme: "light",
    language: "ru",
    notifications: true
};

const userSettings = {
    language: "en",
    fontSize: 14
};

const finalSettings = Object.assign({}, defaults, userSettings);
console.log(finalSettings); 
// {theme: "light", language: "en", notifications: true, fontSize: 14}
```

### Spread-оператор для объектов

```javascript
const defaults = {
    theme: "light",
    language: "ru"
};

const userSettings = {
    language: "en",
    fontSize: 16
};

const combined = {...defaults, ...userSettings};
console.log(combined); // {theme: "light", language: "en", fontSize: 16}

// Добавление новых свойств
const extended = {...combined, lastUpdated: new Date()};
```

## Проверка существования свойств

```javascript
const obj = {
    name: "Виктория",
    age: 29
};

// Оператор in
console.log("name" in obj);     // true
console.log("height" in obj);   // false

// Метод hasOwnProperty
console.log(obj.hasOwnProperty("name"));    // true
console.log(obj.hasOwnProperty("toString")); // false (унаследованное свойство)

// Строгое сравнение с undefined
console.log(obj.name !== undefined);    // true
console.log(obj.height !== undefined);  // false
```

## Перебор свойств объекта

### Цикл for...in

```javascript
const user = {
    name: "Павел",
    age: 31,
    city: "Казань"
};

for (const key in user) {
    if (user.hasOwnProperty(key)) {
        console.log(`${key}: ${user[key]}`);
    }
}
// name: Павел
// age: 31
// city: Казань
```

### Object.keys() с forEach

```javascript
Object.keys(user).forEach(key => {
    console.log(`${key}: ${user[key]}`);
});
```

## Глубокое и поверхностное копирование

### Поверхностное копирование

```javascript
const original = {
    name: "Екатерина",
    address: {
        city: "Ростов-на-Дону",
        street: "Ленина"
    }
};

// Поверхностная копия
const shallowCopy = {...original};
// или
const shallowCopy2 = Object.assign({}, original);

// Изменение вложенного объекта повлияет на обе копии
shallowCopy.address.city = "Воронеж";
console.log(original.address.city); // "Воронеж" - изменилось в оригинале!
```

### Глубокое копирование

```javascript
// Простое глубокое копирование (работает только с JSON-совместимыми данными)
const deepCopy = JSON.parse(JSON.stringify(original));

// Более надежный способ глубокого копирования
function deepClone(obj) {
    if (obj === null || typeof obj !== "object") return obj;
    if (obj instanceof Date) return new Date(obj.getTime());
    if (obj instanceof Array) return obj.reduce((arr, item, i) => {arr[i] = deepClone(item); return arr;}, []);
    if (typeof obj === "object") {
        return Object.keys(obj).reduce((newObj, key) => {
            newObj[key] = deepClone(obj[key]);
            return newObj;
        }, {});
    }
}

const trulyDeepCopy = deepClone(original);
```

## Деструктуризация объектов

```javascript
const employee = {
    firstName: "Анна",
    lastName: "Смирнова",
    position: "Frontend Developer",
    department: "IT",
    salary: 120000,
    contact: {
        email: "anna@example.com",
        phone: "+7 (999) 123-45-67"
    }
};

// Простая деструктуризация
const {firstName, lastName, position} = employee;
console.log(firstName); // "Анна"

// Деструктуризация с переименованием
const {firstName: name, lastName: surname} = employee;
console.log(name); // "Анна"

// Деструктуризация со значениями по умолчанию
const {salary, bonus = 0} = employee;
console.log(bonus); // 0 (значение по умолчанию)

// Деструктуризация вложенных объектов
const {contact: {email, phone}} = employee;
console.log(email); // "anna@example.com"

// Деструктуризация с остаточными параметрами
const {firstName: fName, ...rest} = employee;
console.log(fName); // "Анна"
console.log(rest); // {lastName: "Смирнова", position: "Frontend Developer", ...}
```

## Символы в объектах

```javascript
const userId = Symbol("user id");
const email = Symbol("email");

const user = {
    [userId]: 12345,
    name: "Михаил",
    [email]: "mikhail@example.com"
};

console.log(user[userId]); // 12345
console.log(user[email]);  // "mikhail@example.com"

// Символы не появляются при перечислении
console.log(Object.keys(user)); // ["name"]
console.log(Object.getOwnPropertySymbols(user)); // [Symbol(user id), Symbol(email)]
```

## Практические рекомендации

1. Используйте литералы объектов для создания новых объектов (а не `new Object()`)
2. Используйте деструктуризацию для удобного извлечения значений из объектов
3. Будьте осторожны с поверхностным копированием вложенных объектов
4. Используйте `Object.freeze()` для создания неизменяемых объектов
5. В российских проектах часто используются объекты для конфигурации приложений и хранения состояния

```javascript
// Пример конфигурационного объекта
const appConfig = Object.freeze({
    apiUrl: "https://api.example.com",
    timeout: 5000,
    retries: 3,
    features: {
        analytics: true,
        notifications: false
    }
});
```

> [!tip] Совет
> Используйте `Object.defineProperty()` или `Object.defineProperties()` для создания свойств с особыми характеристиками (например, только для чтения или не перечисляемые).
