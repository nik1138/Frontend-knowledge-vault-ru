# Операторы в JavaScript

Операторы в JavaScript - это символы или ключевые слова, которые выполняют операции над значениями (операндами). Они являются основой для выполнения вычислений, сравнений и логических операций в программировании. В 2025 году понимание операторов критически важно для написания эффективного и понятного кода.

## Арифметические операторы

Арифметические операторы используются для математических вычислений:

```javascript
let a = 10;
let b = 3;

// Основные арифметические операторы
console.log(a + b);  // 13 - сложение
console.log(a - b);  // 7  - вычитание
console.log(a * b);  // 30 - умножение
console.log(a / b);  // 3.333... - деление
console.log(a % b);  // 1  - остаток от деления
console.log(a ** b); // 1000 - возведение в степень (ES2016)

// Унарные операторы
let positive = +5;
let negative = -5;
console.log(positive); // 5
console.log(negative); // -5

// Инкремент и декремент
let counter = 5;
console.log(++counter); // 6 (префиксный инкремент)
console.log(counter++); // 6 (постфиксный инкремент, возвращает старое значение)
console.log(counter);   // 7

let counter2 = 5;
console.log(--counter2); // 4 (префиксный декремент)
console.log(counter2--); // 4 (постфиксный декремент)
console.log(counter2);   // 3
```

## Операторы присваивания

Операторы присваивания присваивают значения переменным:

```javascript
let x = 10; // простое присваивание

// Составные операторы присваивания
x += 5;  // эквивалентно x = x + 5
x -= 3;  // эквивалентно x = x - 3
x *= 2;  // эквивалентно x = x * 2
x /= 4;  // эквивалентно x = x / 4
x %= 3;  // эквивалентно x = x % 3
x **= 2; // эквивалентно x = x ** 2

// Присваивание с логическими операторами (ES2021)
let obj = {};
obj.name ??= "Гость"; // присваивает, только если значение null или undefined
console.log(obj.name); // "Гость"

let settings = { theme: "dark" };
settings.theme &&= "light"; // присваивает "light", если settings.theme истинно
console.log(settings.theme); // "light"

let message = "Привет";
message ||= "Пусто"; // присваивает "Пусто", если message ложно
console.log(message); // "Привет" (не изменилось, так как "Привет" истинно)
```

## Операторы сравнения

Операторы сравнения сравнивают значения и возвращают логическое значение:

```javascript
let num1 = 5;
let num2 = "5";

// Строгое равенство (рекомендуется)
console.log(num1 === num2);  // false (разные типы)
console.log(num1 === 5);     // true

// Нестрогое равенство (не рекомендуется)
console.log(num1 == num2);   // true (происходит преобразование типов)
console.log(0 == false);     // true
console.log("" == false);    // true
console.log(null == undefined); // true

// Строгое неравенство
console.log(num1 !== num2);  // true
console.log(num1 !== 5);     // false

// Другие операторы сравнения
console.log(10 > 5);   // true
console.log(10 < 5);   // false
console.log(10 >= 10); // true
console.log(10 <= 9);  // false

// Сравнение строк
console.log("apple" < "banana"); // true (лексикографическое сравнение)
console.log("Apple" < "banana"); // true (символы в верхнем регистре имеют меньший код)
console.log("2" > "10"); // true (сравнение посимвольно как строки)
console.log(2 > 10);     // false (сравнение как числа)
```

## Логические операторы

Логические операторы используются для комбинации логических значений:

```javascript
// Логическое И (&&)
console.log(true && true);   // true
console.log(true && false);  // false
console.log(false && true);  // false
console.log(false && false); // false

// Короткое замыкание: если первый операнд ложный, второй не вычисляется
let user = null;
let userName = user && user.name; // не вызывает ошибку, возвращает null

// Логическое ИЛИ (||)
console.log(true || true);   // true
console.log(true || false);  // true
console.log(false || true);  // true
console.log(false || false); // false

// Практическое применение: установка значений по умолчанию
function greet(name) {
    name = name || "Гость"; // если name ложно, используем "Гость"
    return `Привет, ${name}!`;
}

console.log(greet()); // "Привет, Гость!"
console.log(greet("Алексей")); // "Привет, Алексей!"

// Логическое НЕ (!)
console.log(!true);  // false
console.log(!false); // true
console.log(!!"hello"); // true (двойное отрицание для преобразования в boolean)

// Новые логические операторы (ES2020)
// Оператор нулевого слияния (??)
let userSetting = null;
let defaultSetting = "темная тема";
let currentSetting = userSetting ?? defaultSetting;
console.log(currentSetting); // "темная тема"

// Отличие от ||: ?? проверяет только null и undefined
let userInput = "";
let output = userInput || "значение по умолчанию";
console.log(output); // "значение по умолчанию" (хотя строка пустая, но не null/undefined)

let output2 = userInput ?? "значение по умолчанию";
console.log(output2); // "" (исходное значение, так как не null/undefined)
```

## Операторы строк

Строки можно объединять с помощью оператора сложения:

```javascript
let firstName = "Анна";
let lastName = "Иванова";
let fullName = firstName + " " + lastName;
console.log(fullName); // "Анна Иванова"

// Сложение числа и строки
let result = 10 + "20"; // "1020" (число преобразуется в строку)
let result2 = "10" + 20; // "1020"
let result3 = 10 + 20 + "5"; // "305" (сначала 10+20=30, затем 30+"5"="305")
let result4 = "5" + 10 + 20; // "51020" (сначала "5"+10="510", затем "510"+20="51020")

// Шаблонные строки (ES6)
let age = 25;
let greeting = `Привет, меня зовут ${fullName} и мне ${age} лет`;
console.log(greeting); // "Привет, меня зовут Анна Иванова и мне 25 лет"
```

## Оператор typeof

Оператор `typeof` возвращает строку, указывающую тип операнда:

```javascript
console.log(typeof 42); // "number"
console.log(typeof "hello"); // "string"
console.log(typeof true); // "boolean"
console.log(typeof undefined); // "undefined"
console.log(typeof {}); // "object"
console.log(typeof []); // "object" (исторический баг)
console.log(typeof function(){}); // "function"
console.log(typeof null); // "object" (исторический баг)
console.log(typeof Symbol()); // "symbol"
console.log(typeof 123n); // "bigint"
```

## Условный (тернарный) оператор

Тернарный оператор предоставляет краткий способ записи условных выражений:

```javascript
let age = 18;
let status = age >= 18 ? "совершеннолетний" : "несовершеннолетний";
console.log(status); // "совершеннолетний"

// Множественные условия
let score = 85;
let grade = score >= 90 ? "A" :
            score >= 80 ? "B" :
            score >= 70 ? "C" :
            score >= 60 ? "D" : "F";
console.log(grade); // "B"

// Практическое применение
function formatPrice(price, currency = "RUB") {
    return price ? `${price} ${currency}` : "Цена не указана";
}

console.log(formatPrice(1000)); // "1000 RUB"
console.log(formatPrice(0)); // "Цена не указана"
```

## Оператор запятая (,)

Оператор запятая вычисляет оба операнда и возвращает значение правого:

```javascript
let a, b, c;
a = 1, b = 2, c = a + b;
console.log(c); // 3

// Часто используется в циклах
for (let i = 0, j = 9; i <= j; i++, j--) {
    console.log(`i: ${i}, j: ${j}`);
}
// Выведет:
// i: 0, j: 9
// i: 1, j: 8
// i: 2, j: 7
// i: 3, j: 6
// i: 4, j: 5
```

## Операторы разложения (Spread) и остаточные операторы (Rest)

### Spread-оператор (...)

```javascript
// Массивы
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6]

// Объекты
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const merged = { ...obj1, ...obj2 }; // { a: 1, b: 2, c: 3, d: 4 }

// Функции
function sum(x, y, z) {
    return x + y + z;
}

const numbers = [1, 2, 3];
console.log(sum(...numbers)); // 6 (эквивалентно sum(1, 2, 3)

// Строки
const chars = [..."hello"]; // ['h', 'e', 'l', 'l', 'o']
```

### Остаточные параметры (Rest)

```javascript
function sumAll(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

console.log(sumAll(1, 2, 3, 4, 5)); // 15
console.log(sumAll(10, 20)); // 30

// Деструктуризация с rest
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first); // 1
console.log(second); // 2
console.log(rest); // [3, 4, 5]
```

## Операторы доступа к свойствам

### Точечная нотация и скобочная нотация

```javascript
const obj = {
    name: "Мария",
    "full name": "Мария Петрова",
    123: "числовой ключ"
};

console.log(obj.name); // "Мария" - точечная нотация
console.log(obj["full name"]); // "Мария Петрова" - скобочная нотация
console.log(obj[123]); // "числовой ключ" - или obj["123"]
console.log(obj["name"]); // "Мария" - также работает

// Динамический доступ
const propertyName = "name";
console.log(obj[propertyName]); // "Мария"
```

## Современные операторы (ES2020+)

### Optional chaining (?.)

```javascript
const user = {
    name: "Алексей",
    address: {
        city: "Москва",
        street: "Тверская"
    }
};

// Безопасный доступ к вложенным свойствам
console.log(user.address?.city); // "Москва"
console.log(user.address?.country?.iso); // undefined (без ошибки)
console.log(user.profile?.avatar); // undefined (без ошибки)

// Вызов методов
const safeUser = null;
console.log(safeUser?.getName?.()); // undefined (без ошибки)

// Доступ к элементам массива
const items = [1, 2, 3];
console.log(items?.[0]); // 1
console.log(items?.[10]?.toString()); // undefined
```

### Nullish coalescing (??)

```javascript
// ?? отличается от || тем, что проверяет только null и undefined
const response = {
    name: "Иван",
    age: 0, // это может быть валидным значением
    city: "", // это может быть валидным значением
    active: false // это может быть валидным значением
};

// Использование || может привести к нежелательному поведению
const displayAge = response.age || 18; // 18, хотя 0 может быть валидным
const displayCity = response.city || "Не указан"; // "Не указан", хотя пустая строка может быть валидной

// Использование ?? для более точного поведения
const actualAge = response.age ?? 18; // 0 (правильное значение)
const actualCity = response.city ?? "Не указан"; // "" (правильное значение)
const actualActive = response.active ?? true; // false (правильное значение)

console.log({ actualAge, actualCity, actualActive });
// { actualAge: 0, actualCity: "", actualActive: false }
```

## Практические примеры

### Валидация данных с операторами

```javascript
function validateUser(user) {
    // Проверка, что все обязательные поля присутствуют
    const isValid = user && 
                   user.name && 
                   user.email && 
                   user.email.includes('@') &&
                   user.age !== undefined && 
                   user.age >= 18;
    
    return isValid;
}

// Использование логических операторов для безопасного доступа
function getUserDisplayName(user) {
    return user?.profile?.name || user?.name || "Анонимный пользователь";
}

const user1 = { name: "Анна", email: "anna@example.com", age: 25 };
const user2 = { name: "Борис" }; // без email и age

console.log(validateUser(user1)); // true
console.log(validateUser(user2)); // false
console.log(getUserDisplayName(user1)); // "Анна"
console.log(getUserDisplayName({ profile: { name: "Виктор" } })); // "Виктор"
```

### Работа с API и операторами

```javascript
// Функция для обработки ответа от API
function processApiResponse(response) {
    // Используем optional chaining и nullish coalescing для безопасной обработки
    const data = response?.data ?? [];
    const total = response?.pagination?.total ?? 0;
    const hasMore = response?.pagination?.hasMore ?? false;
    
    return {
        items: data,
        total,
        hasMore,
        // Устанавливаем статус по умолчанию, если его нет
        status: response?.status ?? 'unknown'
    };
}

// Примеры использования
const response1 = { 
    data: [{ id: 1, name: "Товар 1" }], 
    pagination: { total: 1, hasMore: false } 
};

const response2 = {}; // частичный или пустой ответ

console.log(processApiResponse(response1));
// { items: [{ id: 1, name: "Товар 1" }], total: 1, hasMore: false, status: 'unknown' }

console.log(processApiResponse(response2));
// { items: [], total: 0, hasMore: false, status: 'unknown' }
```

## Практические рекомендации

1. Используйте строгое равенство (`===`) вместо нестрогого (`==`) для предотвращения неожиданных преобразований типов
2. Используйте оператор нулевого слияния (`??`) вместо логического ИЛИ (`||`) когда нужно отличать `null`/`undefined` от других ложных значений
3. Используйте optional chaining (?.) для безопасного доступа к вложенным свойствам
4. В российских проектах часто используется комбинация операторов для проверки и установки значений по умолчанию
5. Используйте тернарный оператор для простых условий, но избегайте его чрезмерного вложения

```javascript
// Хорошая практика: комбинация операторов для безопасной обработки данных
function getConfigValue(config, path, defaultValue = null) {
    // Разбиваем путь на части и последовательно проходим по объекту
    const pathParts = path.split('.');
    let current = config;
    
    for (const part of pathParts) {
        current = current?.[part];
        if (current === undefined || current === null) {
            break;
        }
    }
    
    return current ?? defaultValue;
}

const appConfig = {
    api: {
        baseUrl: "https://api.example.com",
        timeout: 5000
    },
    ui: {
        theme: "dark"
    }
};

console.log(getConfigValue(appConfig, "api.baseUrl", "https://default.com"));
// "https://api.example.com"
console.log(getConfigValue(appConfig, "database.host", "localhost"));
// "localhost"
```

> [!tip] Совет
> Используйте операторы последовательного вычисления (&&, ||, ??) для создания более читаемого и компактного кода, особенно при обработке условий и установке значений по умолчанию.
